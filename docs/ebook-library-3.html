#file: providers/goodreads/scraper.py
import functions as fn
import re

ratings_regex = re.compile(r'([0-9]*\.[0-9]*) avg rating')

def goodreads_parse_book(book_soup):
    img = book_soup.findAll(itemprop='image')[0]['src']
    title = book_soup.findAll(class_='bookTitle')[0].get_text().strip()
    authors = book_soup.findAll(class_='authorName')
    authors = [a.get_text().strip() for a in authors]
    rating = re.search(ratings_regex, str(book_soup))[1]
    return { 'img':img,
             'title':title,
             'authors':authors,
             'rating':rating }

def goodreads_results(url):
    # Fetch html webpage and create BeautifulSoup object
    html_content = fn.get_html(url)
    soup = fn.get_soup(html_content)

    # find all book rows
    books_html = soup.findAll(itemtype='http://schema.org/Book')

    # parse each row
    books = [goodreads_parse_book(html) for html in books_html]
    return books

def submit_query(tokens):
    url = 'https://www.goodreads.com/search?utf8=&q=' + '+'.join(tokens) + '&search_type=books'
    return goodreads_results(url)

There only remains to format the results using the html_presenter. We will reuse the template_loader from before.

#file: providers/goodreads/html_presenter.py
from io import StringIO
from template_loader import load_template

def format(results):
    content = StringIO()
    content.write('<table>')
    template = load_template('item.html', './providers/goodreads/')
    for item_id, result in enumerate(results):
        content.write(template.substitute(
            img = result.get('img', ''),
            title = result.get('title', ''),
            authors = ' - '.join(result.get('authors', [])),
            rating = result.get('rating', ''),
            ))
    content.write('</table>')
    return content.getvalue()
